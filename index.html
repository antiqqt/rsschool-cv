            function perimeter(n) {

            function fib(n) {
              let [a, b] = [0, 1];
              let temp = [];
              for (let i = 0; i < n + 1; i++) {
                b = a + b; 
                a = b - a;
                temp.push(a);
              }
              return temp.reduce((acc, v) => acc + v);
            }
